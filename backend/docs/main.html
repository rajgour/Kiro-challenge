470@app.post("/events/{event_id}/registrations", status_code=status.HTTP_201_CREATED)
471def register_user_for_event(event_id: str, registration: RegistrationRequest):
472    if not event_id or len(event_id) > 100:
473        raise HTTPException(
474            status_code=status.HTTP_400_BAD_REQUEST,
475            detail="Invalid event ID"
476        )
477    
478    user_id = registration.userId
479    
480    try:
481        # Check if event exists
482        event_response = table.get_item(Key={'eventId': event_id})
483        if 'Item' not in event_response:
484            logger.warning(f"Event not found: {event_id}")
485            raise HTTPException(
486                status_code=status.HTTP_404_NOT_FOUND,
487                detail="Event not found"
488            )
489        event = event_response['Item']
490        
491        # Check if user exists
492        user_response = users_table.get_item(Key={'userId': user_id})
493        if 'Item' not in user_response:
494            logger.warning(f"User not found: {user_id}")
495            raise HTTPException(
496                status_code=status.HTTP_404_NOT_FOUND,
497                detail="User not found"
498            )
499        
500        # Check if already registered
501        existing_reg = registrations_table.get_item(
502            Key={'eventId': event_id, 'userId': user_id}
503        )
504        if 'Item' in existing_reg:
505            raise HTTPException(
506                status_code=status.HTTP_409_CONFLICT,
507                detail="Already registered for this event"
508            )
509        
510        # Get current counts
511        registered_count = event.get('registeredCount', 0)
512        waitlist_count = event.get('waitlistCount', 0)
513        capacity = event.get('capacity', 0)
514        waitlist_enabled = event.get('waitlistEnabled', False)
515        
516        # Determine registration status
517        if registered_count < capacity:
518            # Space available - register directly
519            reg_status = "registered"
520            waitlist_position = None
521            new_registered_count = registered_count + 1
522            new_waitlist_count = waitlist_count
523        elif waitlist_enabled:
524            # Event full but waitlist enabled
525            reg_status = "waitlisted"
526            waitlist_position = waitlist_count + 1
527            new_registered_count = registered_count
528            new_waitlist_count = waitlist_count + 1
529        else:
530            # Event full and no waitlist
531            raise HTTPException(
532                status_code=status.HTTP_409_CONFLICT,
533                detail="Event is full"
534            )
535        
536        # Create registration
537        registration_item = {
538            'eventId': event_id,
539            'userId': user_id,
540            'status': reg_status,
541            'registeredAt': datetime.utcnow().isoformat()
542        }
543        if waitlist_position is not None:
544            registration_item['waitlistPosition'] = waitlist_position
545        
546        registrations_table.put_item(Item=registration_item)
547        
548        # Update event counts
549        table.update_item(
550            Key={'eventId': event_id},
551            UpdateExpression='SET registeredCount = :rc, waitlistCount = :wc',
552            ExpressionAttributeValues={
553                ':rc': new_registered_count,
554                ':wc': new_waitlist_count
555            }
556        )
557        
558        logger.info(f"Registered user {user_id} for event {event_id} with status {reg_status}")
559        return registration_item
560        
561    except HTTPException:
562        raise
563    except ClientError as e:
564        logger.error(f"DynamoDB error: {e.response['Error']['Message']}")
565        raise HTTPException(
566            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
567            detail="Failed to register user for event"
568        )
